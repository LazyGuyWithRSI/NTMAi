// Gib Base
class NTM_Gib : Actor {
	Default {
		FloatBobPhase 0;
		Health 10;
		Radius 4;
		Height 8;
		Mass 1;
		Gravity 0.5;
		BounceType "Doom";
		
		+Corpse
		+NoTeleport
		+NoBlockMap
		+NotOnAutomap
		+BounceOnWalls
		+DontSplash
		+ThruActors
		+FloorClip
	}
	
	States {
		Spawn:
		Crash:
			TNT1 A 0;
			Stop;
	}
	
	override void PostBeginPlay() {
		Super.PostBeginPlay();
		scale.x *= FRandomPick(-1, 1);
		scale.y *= FRandomPick(-1, 1);
	}
}

}

// Bullet Casing
class NTM_BulletCasing : NTM_Gib {
	Default {
		Scale 0.16;
		Radius 1;
		Height 1;
		Gravity 1;
		BounceSound "NTM_BulletCasing/bounce";
		BounceType "Doom";
		
		-Corpse
		+CanBounceWater
		+Missile
		+UseBounceState
	}
	
	States
	{
		Spawn:
			NBUL A 0;
		Roll:
			#### ABCDEFGH 2;
			Loop;
			
		Bounce:
			#### A 0;
			Goto Roll;
			
		Death:
			#### C 350 A_Jump(128, 1);
			Stop;
			#### G 350;
			Stop;
	}
}
